<script>
  document.addEventListener("DOMContentLoaded", function () {
    const content = document.getElementById("project-index-content");
    const nav = document.getElementById("project-timeline-nav");

    if (!content || !nav) {
      return;
    }

    const slugify = function (text) {
      return text
        .toLowerCase()
        .replace(/[^a-z0-9]+/g, "-")
        .replace(/^-+|-+$/g, "");
    };

    const timeline = [];
    let currentYear = null;

    content.querySelectorAll("h2, .card.hoverable").forEach(function (node) {
      if (node.tagName === "H2" && /^20\d{2}$/.test(node.textContent.trim())) {
        const year = node.textContent.trim();
        node.id = node.id || "projects-" + year;
        currentYear = { year: year, target: node.id, projects: [] };
        timeline.push(currentYear);
        return;
      }

      if (!currentYear || !node.classList.contains("card")) {
        return;
      }

      const titleNode = node.querySelector(".card-title");
      if (!titleNode) {
        return;
      }

      const title = titleNode.textContent.trim();
      const row = node.closest(".row") || node;
      row.id = row.id || "project-" + slugify(title);
      currentYear.projects.push({ title: title, target: row.id });
    });

    nav.innerHTML = timeline
      .map(function (group) {
        const projects = group.projects
          .map(function (project) {
            return '<a class="project-timeline-link" href="#' + project.target + '">' + project.title + "</a>";
          })
          .join("");

        return (
          '<section class="project-timeline-year">' +
          '<a class="project-timeline-year-link" href="#' + group.target + '">' + group.year + "</a>" +
          '<div class="project-timeline-projects">' + projects + "</div>" +
          "</section>"
        );
      })
      .join("");

    nav.addEventListener("click", function (event) {
      const link = event.target.closest("a[href^='#']");
      if (!link) {
        return;
      }

      const target = document.querySelector(link.getAttribute("href"));
      if (!target) {
        return;
      }

      event.preventDefault();
      target.scrollIntoView({ behavior: "smooth", block: "start" });
      history.replaceState(null, "", link.getAttribute("href"));
    });
  });
</script>
